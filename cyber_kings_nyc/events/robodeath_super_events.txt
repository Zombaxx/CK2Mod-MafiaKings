#AI triggered automatically

character_event = {

	id = mechfl.3299
	desc = "EVTDESCmechfl.2"
	title = "EVTNAMEmechfl.2"
	is_triggered_only = yes
	hide_window = yes
	
	trigger = {
		war = yes
		has_artifact = super_mech_suit
		in_command = yes
		ai = yes
		NOT = { trait = maimed }
		NOT = { trait = wounded }
		NOT = { has_character_modifier = usedmech }
	}

	option = {
		name = mechfl.atk4
		ai_chance = {factor = 100}
		if = {
		
			martial <= 15
			limit = {NOT = {martial >= 15}}
			any_current_enemy = {
				limit = { in_command = yes at_location = ROOT}
				troops = -0.20
				}
			any_current_enemy = {
				any_realm_lord = {
					limit = { in_command = yes at_location = ROOT}
					troops = -0.10
					}
				any_courtier = {
					limit = { in_command = yes at_location = ROOT}
					troops = -0.15
					}
			}
		}		
		
		if = {
		
			martial >= 18
			any_current_enemy = {
				limit = { in_command = yes at_location = ROOT}
				troops = -0.45
				}
			any_current_enemy = {
				any_realm_lord = {
					limit = { in_command = yes at_location = ROOT}
					troops = -0.15
					}
				any_courtier = {
					limit = { in_command = yes at_location = ROOT}
					troops = -0.25
				}
			}
		}
		
		if = {
		
			limit = {trait = wounded }
			
			random = {
				chance = 50
				add_trait = maimed
				
			}
		
		}
		
		if = {
			limit = { NOT = { trait = maimed } NOT = { trait = wounded } }
			random = {
				chance = 15
				add_trait = wounded
			}
				
		}
		
		hidden_tooltip = {
				add_character_modifier = { modifier = usedmech duration = 30}
			} 
	}
}
